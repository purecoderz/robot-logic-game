package main

func turnRight() {
    turnLeft()
    turnLeft()
    turnLeft()
}

func NextRow() {
    for i := 0; i < 9; i++ {
        move()
    }

}

func jumpLeft() {
    turnLeft()
    move()
    turnLeft()
}


func jumpRight() {
    turnRight()
    move()
    turnRight()
}



func main() {

    for i := 1; i <= 9; i++ {
        NextRow()
        if i%2 == 0 {
           jumpRight()
        }else{
           jumpLeft()
        }
        
    }

}
